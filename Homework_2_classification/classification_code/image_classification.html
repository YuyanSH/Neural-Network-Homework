import torch.nn as nn
# import torch.nn.functional as F

class CNN(nn.Module):
    def __init__(self):
        super(CNN, self).__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 64, kernel_size=3, padding=1), # 13643x3padding=1
            nn.BatchNorm2d(64), # 
            nn.ReLU(inplace=True), # ReLU
            nn.MaxPool2d(2),  # 2x2
            nn.Conv2d(64, 128, kernel_size=3, padding=1), # 264128
            nn.BatchNorm2d(128), # 
            nn.ReLU(inplace=True), # ReLU
            nn.MaxPool2d(2),  # 2x2
            nn.Conv2d(128, 256, kernel_size=3, padding=1), # 3128256
            nn.BatchNorm2d(256), # 
            nn.ReLU(inplace=True), # ReLU
            nn.MaxPool2d(2),  # 2x2
        )
        self.classifier = nn.Sequential(
            nn.Dropout(0.5), # Dropout
            nn.Linear(256*4*4, 1024), # 1256*4*4=40961024
            nn.ReLU(inplace=True), # ReLU
            nn.Dropout(0.5), # Dropout
            nn.Linear(1024, 10), # 2102410CIFAR-1010
        )

    def forward(self, x):
        x = self.features(x) # 
        x = x.view(x.size(0), -1) # -1
        x = self.classifier(x) # 
        return x
    
# 
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
net = CNN().to(device)

def train(epoch):
    net.train() # 
    running_loss = 0.0 # 

    # 
    for batch_idx, (inputs, targets) in enumerate(train_loader):
        inputs, targets = inputs.to(device), targets.to(device) # GPU
        optimizer.zero_grad() # ()
        outputs = net(inputs) # ()
        loss = criterion(outputs, targets) # 
        loss.backward() # ()
        optimizer.step() # 
        running_loss += loss.item() # 
    print(f'Epoch {epoch}: Loss = {running_loss/len(train_loader):.4f}') # epoch

from sklearn.metrics import precision_score
def evaluate():
    net.eval() # 
    correct = 0 # 
    total = 0 # 
    all_preds = [] # 
    all_targets = [] # 
    
    # 
    # torch.no_grad()
    with torch.no_grad():
        for inputs, targets in test_loader: # test_loader
            inputs, targets = inputs.to(device), targets.to(device) # GPU
            outputs = net(inputs) # ()
            _, predicted = outputs.max(1) # max(1)
            total += targets.size(0) # 
            correct += (predicted == targets).sum().item() # 
            all_preds.extend(predicted.cpu().numpy()) # GPUCPUnumpy
            all_targets.extend(targets.cpu().numpy()) # GPUCPUnumpy
    acc = correct / total # 
    precisions = precision_score(all_targets, all_preds, average=None) # 
    print(f'Test Accuracy: {acc:.4f}')
    for i, p in enumerate(precisions):
        print(f'Class {i} Precision: {p:.4f}')
    return acc, precisions #
